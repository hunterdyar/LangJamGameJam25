
{on-spawn
	(set x 0)
	(set y 0)
	(register-input-event on-left left press)
	(register-input-event on-left a press)
	(register-input-event on-right right press)
	(register-input-event on-right d press)
	(set g (find-comp-in-parent grid))
}

{render
	(draw-grid-sprite :g (get x) (get y) slime)
}

{on-left
 (if (gt :x 0)
	(set x (dec :x))
  )
}
{on-right
	(if (lt :x (get-from :g cols))
		(set x (inc :x))
	)
}
